\begin{problem}
    Which of the following sequences converge?
	\begin{enumerate}
		\item[P.] \(\left \{ \dfrac{(-1)^n(n+1)}{n}\right \}\)
	
		\item[Q.] \( \left \{ \dfrac{(-1)^n}{n} \right \} \)
		
		\item[R.] \(\left \{ \dfrac{1}{n} \right \} \)
	\end{enumerate}
\begin{multipleChoice}
	\choice{P only}
	\choice{Q only}
	\choice{P, Q, and R}
	\choice{P and Q only}
	\choice[correct]{Q and R only} %correct
\end{multipleChoice}
\end{problem}
